import itertools
from collections import deque
import sys
sys.stdin = open('test.txt', 'r')

T = int(input())
for tc in range(1, T + 1):
    N = int(input())
    weight_cho = list(map(int, input().split()))
    cnt = 0
    def generate_subset(depth, included):
        global left_sum, right_sum, cnt
        if depth == N:
            left_sum = 0
            right_sum = 0
            for i in range(N):
                if included[i]:
                    left_sum += cho[i]
                else:
                    right_sum += cho[i]
                    if left_sum < right_sum:
                        return
            else:
                cnt += 1
                return

        included[depth] = False
        generate_subset(depth + 1, included)

        included[depth] = True
        generate_subset(depth + 1, included)

    for cho in itertools.permutations(weight_cho, N):
        init_included = [False] * N
        init_included[0] = True
        left_sum = 0
        right_sum = 0
        generate_subset(1, init_included)

    print(f'#{tc} {cnt}')
