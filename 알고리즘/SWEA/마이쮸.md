### 마이쮸
```
total_candy = 20    # 총 주어진 마이쮸 개수
queue = []  # 사람들이 줄 서는 큐

next_person = 1
# 누가 받을지
# 몇 개 받을지
queue.append((next_person, 1))
last_person = None # 마지막으로 마이쮸 받을 사람

while total_candy > 0:
    person, cnt = queue.pop(0)

    if total_candy - cnt <= 0:
        last_person = person
        break

    total_candy -= cnt

    queue.append((person, cnt + 1))


    next_person += 1
    queue.append((next_person, 1))

print("마지막 마이쮸 사람: ", last_person)
```
